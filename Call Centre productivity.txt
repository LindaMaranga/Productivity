 SELECT

T.INVOICEID,

T.CAPTUREDBY,

T.ASSESSEDBY,T.InvoiceReference,

T.CAPTUREDATE,

T.ASSESSEDDATE,

P.STATUSCODEDESC,

T.INVOICEBENEFIT,

T.INVOICELINEDETAILS,

T.PBC,

P.PREAUTHORISATIONDATE

FROM

MIDBCLAIMTREATMENTINVOICE T

INNER JOIN MIDBPREAUTHORISEDINVOICE P ON P.INVOICEID=T.INVOICEID

 

WHERE

PBC = 'KES'

AND

ASSESSEDDATE >='2023-10-01'
